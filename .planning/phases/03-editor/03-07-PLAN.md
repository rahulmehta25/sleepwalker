After:   OPENAI_API_KEY=\${OPENAI_API_KEY}`}

    "use client";
    import { useEffect, useState } from "react";

    interface DraftFields {
      name: string; slug: string; runtime: string; prompt: string; schedule: string; reversibility: string; budget: number | string;
    }
    interface StoredDraft {
      version: number;
      updatedAt: string;
      fields: DraftFields;
    }

    interface Props {
      onRestore: (d: DraftFields) => void;
      onStartFresh: () => void;
    }

    const KEY = "sleepwalker.draft.v1";

    function relativeTime(iso: string): string {
      const t = Date.parse(iso);
      if (Number.isNaN(t)) return "recently";
      const mins = Math.max(1, Math.floor((Date.now() - t) / 60_000));
      if (mins < 60) return `${mins} minute${mins === 1 ? "" : "s"} ago`;
      const hours = Math.floor(mins / 60);
      if (hours < 24) return `${hours} hour${hours === 1 ? "" : "s"} ago`;
      const days = Math.floor(hours / 24);
      return `${days} day${days === 1 ? "" : "s"} ago`;
    }

    export function DraftRecoveryBanner({ onRestore, onStartFresh }: Props): JSX.Element | null {
      const [draft, setDraft] = useState<StoredDraft | null>(null);
      const [dismissed, setDismissed] = useState(false);

      useEffect(() => {
        if (typeof window === "undefined") return;
        const raw = localStorage.getItem(KEY);
        if (!raw) return;
        try {
          const parsed = JSON.parse(raw);
          if (parsed && parsed.version === 1 && parsed.fields) {
            setDraft(parsed as StoredDraft);
          }
        } catch {
          // malformed — silently ignore
        }
      }, []);

      if (!draft || dismissed) return null;

      const handleRestore = () => {
        onRestore(draft.fields);
        setDismissed(true);
      };
      const handleStartFresh = () => {
        try { localStorage.removeItem(KEY); } catch {/* ignore */}
        onStartFresh();
        setDismissed(true);
      };

      return (
        <div className="panel-raised border border-aurora-400/30 p-4 mb-6 flex items-center justify-between gap-4">
          <div className="flex flex-col gap-1">
            <p className="text-sm">You have an unsaved draft from {relativeTime(draft.updatedAt)}.</p>
            <p className="text-xs text-moon-400">
              {draft.fields.name || "Unnamed routine"} · {draft.fields.runtime || "no runtime"} · {draft.fields.slug || "no slug"}
            </p>
          </div>
          <div className="flex gap-2">
            <button type="button" className="btn-ghost text-aurora-400" onClick={handleRestore}>Restore draft</button>
            <button type="button" className="btn-ghost" onClick={handleStartFresh}>Start fresh</button>
          </div>
        </div>
      );
    }
    ```

    "use client";
    import type { HealthStatus, Runtime } from "@/lib/runtime-adapters/types";
    import { CronPreview } from "./cron-preview";

    interface Props {
      runtime: Runtime | "";
      slug: string;
      schedule: string;
      healthStatus: HealthStatus | null;
    }

    function safe<T>(fn: () => T, fallback: T): T {
      try { return fn(); } catch { return fallback; }
    }

    function bundlePath(runtime: Runtime | "", slug: string): string {
      if (!runtime || !slug) return "—";
      const root = ({
        "claude-routines": "routines-cloud",
        "claude-desktop":  "routines-local",
        "codex":           "routines-codex",
        "gemini":          "routines-gemini",
      } as const)[runtime];
      return `${root}/${slug}/`;
    }

    function plistPath(runtime: Runtime | "", slug: string): string | null {
      if (runtime !== "codex" && runtime !== "gemini") return null;
      if (!slug) return null;
      return safe(() => `~/Library/LaunchAgents/com.sleepwalker.${runtime}.${slug}.plist`, "—");
    }

    function markerTag(runtime: Runtime | "", slug: string): string {
      if (!runtime || !slug) return "—";
      return `[sleepwalker:${runtime}/${slug}]`;
    }

    export function PreviewPanel({ runtime, slug, schedule, healthStatus }: Props): JSX.Element {
      return (
        <aside className="w-80 sticky top-10 panel p-4 flex flex-col gap-3">
          <div>
            <div className="label mb-1">BUNDLE PATH</div>
            <div className="data text-xs text-moon-400">{bundlePath(runtime, slug)}</div>
          </div>
          {plistPath(runtime, slug) && (
            <div>
              <div className="label mb-1">PLIST PATH</div>
              <div className="data text-xs text-moon-600">{plistPath(runtime, slug)}</div>
            </div>
          )}
          <div>
            <div className="label mb-1">MARKER TAG</div>
            <div className="data text-xs text-moon-600">{markerTag(runtime, slug)}</div>
          </div>
          <div>
            <div className="label mb-1">SCHEDULE</div>
            <CronPreview expression={schedule} />
          </div>
          {healthStatus && (
            <div>
              <div className="label mb-1">HEALTH</div>
              <div className="text-xs">
                {healthStatus.available
                  ? <span className="pill-green">Ready</span>
                  : <span className="pill-amber">{healthStatus.reason ?? "Unavailable"}</span>}
              </div>
            </div>
          )}
        </aside>
      );
    }
    ```

    NOTE: Using a local runtime→root map instead of importing toBundleDir because toBundleDir throws on invalid slug (per Phase 2 CONTEXT.md line 43), and PreviewPanel must render mid-keystroke safely. The `safe()` wrapper is kept on plistPath for consistency even though the current template is string-interp only — future-proofing for when a real builder is wired.
