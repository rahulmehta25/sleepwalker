       import fs from "node:fs";
       import os from "node:os";
       import path from "node:path";
       import crypto from "node:crypto";
       import { simpleGit, type SimpleGit } from "simple-git";
       import * as lockfile from "proper-lockfile";
       import { RUNTIME_ROOT } from "@/lib/bundles";
       import type { Runtime } from "@/lib/runtime-adapters/types";

       // Result types (see <interfaces> block)
       export interface PreviewResult { /* ... */ }
       export type SaveToRepoError = /* ... */;
       export interface CommitResult { /* ... */ }

       // Lock sentinel path
       function sleepwalkerDir(): string {
         return path.join(process.env.HOME || os.homedir(), ".sleepwalker");
       }
       function gitLockSentinel(): string {
         return path.join(sleepwalkerDir(), "git.lock.sentinel");
       }

       // Resolve repo root. dashboard/ cwd → parent = repo root.
       // Override via SLEEPWALKER_REPO_ROOT env var (used by tests).
       function repoRoot(): string {
         return process.env.SLEEPWALKER_REPO_ROOT || path.resolve(process.cwd(), "..");
       }

       // Module-scope registry holding release callbacks between preview and commit.
       // See 04-RESEARCH.md Assumption A3 — Sleepwalker is single-process; this is safe.
       const LOCK_REGISTRY = new Map<string, {
         release: () => Promise<void>;
         runtime: Runtime;
         slug: string;
         startedAt: number;
       }>();

       export async function previewSaveToRepo(runtime: Runtime, slug: string): Promise<PreviewResult | SaveToRepoError> {
         // Touch sentinel
         await fs.promises.mkdir(sleepwalkerDir(), { recursive: true });
         const sentinel = gitLockSentinel();
         try { await fs.promises.writeFile(sentinel, "", { flag: "a" }); } catch {}

         // Acquire lock — non-blocking
         let release: () => Promise<void>;
         try {
           release = await lockfile.lock(sentinel, { retries: 0, stale: 30_000 });
         } catch (e: any) {
           if (e && e.code === "ELOCKED") {
             return { ok: false, kind: "lock-busy", error: "Another save-to-repo is in progress. Wait a moment and try again." };
           }
           return { ok: false, kind: "git-error", error: e?.message ?? String(e) };
         }

         try {
           const git: SimpleGit = simpleGit(repoRoot());
           const subpath = `${RUNTIME_ROOT[runtime]}/${slug}/`;
           await git.add(["--", subpath]);
           const diff = await git.diffSummary(["--cached", "--", subpath]);

           if (diff.files.length === 0) {
             // Nothing staged; clean up and release.
             try { await git.raw(["reset", "HEAD", "--", subpath]); } catch {}
             await release();
             return { ok: false, kind: "no-changes", error: "No changes to commit — bundle is in sync with HEAD." };
           }

           // Verb inference: if any prior commit touched this subpath → update, else add.
           let verb: "add" | "update" = "add";
           try {
             const log = await git.raw(["log", "--oneline", "--", subpath]);
             if (log.trim().length > 0) verb = "update";
           } catch {}

           const suggestedMessage = `feat(routines): ${verb} ${runtime}/${slug}`;
           const lockToken = crypto.randomBytes(16).toString("hex");
           LOCK_REGISTRY.set(lockToken, { release, runtime, slug, startedAt: Date.now() });

           return {
             ok: true,
             files: diff.files.map((f) => ({
               path: f.file,
               added: "insertions" in f ? (f.insertions as number) : 0,
               removed: "deletions" in f ? (f.deletions as number) : 0,
             })),
             totals: { filesChanged: diff.files.length, added: diff.insertions, removed: diff.deletions },
             suggestedMessage,
             lockToken,
           };
         } catch (e: any) {
           await release().catch(() => {});
           return { ok: false, kind: "git-error", error: e?.message ?? String(e) };
         }
       }

       export async function commitSaveToRepo(args: { lockToken: string; message: string }): Promise<CommitResult | { ok: false; error: string }> {
         const entry = LOCK_REGISTRY.get(args.lockToken);
         if (!entry) return { ok: false, error: "Lock expired. Reopen Save to repo." };
         try {
           const git = simpleGit(repoRoot());
           const commit = await git.commit(args.message);
           LOCK_REGISTRY.delete(args.lockToken);
           await entry.release();
           return { ok: true, sha: commit.commit, shortSha: commit.commit.slice(0, 7) };
         } catch (e: any) {
           return { ok: false, error: e?.message ?? String(e) };
         }
       }

       export async function releaseSaveLock(args: { lockToken: string }): Promise<{ ok: true }> {
         const entry = LOCK_REGISTRY.get(args.lockToken);
         if (!entry) return { ok: true };
         try {
           const git = simpleGit(repoRoot());
           const subpath = `${RUNTIME_ROOT[entry.runtime]}/${entry.slug}/`;
           // Per Pitfall #7: git rm --cached handles both tracked (no-op-ish) and newly-added paths.
           try { await git.raw(["rm", "--cached", "--ignore-unmatch", "-r", "--", subpath]); } catch {}
           try { await git.raw(["reset", "HEAD", "--", subpath]); } catch {}
         } catch {}
         LOCK_REGISTRY.delete(args.lockToken);
         await entry.release().catch(() => {});
         return { ok: true };
       }
