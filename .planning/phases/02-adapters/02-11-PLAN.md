```typescript
// dashboard/lib/runtime-adapters/supervisor-staging.ts
//
// Content-hash-versioned staging of bin/sleepwalker-run-cli into
// ~/.sleepwalker/bin/sleepwalker-run-cli-<hash8>. Two problems solved:
//
// 1. macOS TCC blocks launchd from executing binaries in protected paths
//    (~/Desktop, ~/Documents, ~/Downloads, iCloud). Staging into
//    ~/.sleepwalker/ (user's home, not TCC-protected) makes the executable
//    half of the supervisor pipeline work regardless of where the repo lives.
//
// 2. Overwriting an executing supervisor binary mid-run is defined but
//    fragile on macOS. Versioned filenames (one-per-content-hash) mean a
//    concurrent re-stage NEVER replaces the file a live launchd job holds.
//
// Source path resolution mirrors codex.ts/gemini.ts: process.env
// SLEEPWALKER_REPO_ROOT override with process.cwd()+/.. fallback.

import crypto from "node:crypto";
import fs from "node:fs/promises";
import os from "node:os";
import path from "node:path";

function repoRoot(): string {
  return process.env.SLEEPWALKER_REPO_ROOT || path.resolve(process.cwd(), "..");
}

function sourcePath(): string {
  return path.join(repoRoot(), "bin", "sleepwalker-run-cli");
}

function stagingDir(): string {
  const home = process.env.HOME || os.homedir();
  return path.join(home, ".sleepwalker", "bin");
}

async function sha256Hex8(filePath: string): Promise<string> {
  const bytes = await fs.readFile(filePath);
  return crypto.createHash("sha256").update(bytes).digest("hex").slice(0, 8);
}

/**
 * Ensure a content-versioned copy of bin/sleepwalker-run-cli exists in
 * ~/.sleepwalker/bin/. Returns the absolute path to the staged copy.
 *
 * Idempotent: second call with unchanged source is a stat + hash only
 * (sub-millisecond). fs.copyFile is atomic (POSIX rename semantics); mode
 * is explicitly re-applied via fs.chmod because fs.copyFile's preserve-mode
 * behavior varies across Node versions.
 *
 * Throws if source is missing — caller's deploy() wraps in try/catch and
 * surfaces the error to DeployResult.
 */
export async function ensureStagedSupervisor(): Promise<string> {
  const src = sourcePath();
  // fs.stat first for the clearest error path — readFile on missing file
  // buries the path in a generic ENOENT message.
  let srcStat;
  try {
    srcStat = await fs.stat(src);
  } catch (e) {
    throw new Error(
      `Supervisor source not found at ${src}. ` +
        `Set SLEEPWALKER_REPO_ROOT to the repo root, or run install.sh first.`,
    );
  }
  if (!srcStat.isFile()) {
    throw new Error(`Supervisor source at ${src} is not a regular file.`);
  }

  const hash = await sha256Hex8(src);
  const destDir = stagingDir();
  const dest = path.join(destDir, `sleepwalker-run-cli-${hash}`);

  // Fast path: staged copy with matching hash already exists and is
  // executable — return it. We trust the filename's hash to be honest.
  try {
    const destStat = await fs.stat(dest);
    if (destStat.isFile() && (destStat.mode & 0o111) !== 0) {
      return dest;
    }
  } catch {
    // ENOENT is expected on first stage; fall through to copy
  }

  await fs.mkdir(destDir, { recursive: true });
  // Defense-in-depth: unlink any pre-existing partial write. fs.copyFile
  // with the default flags overwrites, but an unlink-first makes the
  // intent explicit and survives weird filesystem corner cases.
  await fs.unlink(dest).catch(() => undefined);
  await fs.copyFile(src, dest);
  // Explicit chmod because fs.cp's mode-preservation varies across
  // Node versions and fs.copyFile does not preserve mode on some
  // platforms. 0o755 = rwxr-xr-x, required for launchd exec.
  await fs.chmod(dest, 0o755);

  return dest;
}
```

Key design points:
1. Hash is first 8 hex chars of sha256 — 4 billion values, collision-unlikely for the realistic number of supervisor revisions (dozens over project lifetime).
2. `fs.copyFile` + explicit `fs.chmod 0o755` (NOT `fs.cp`, whose preserve-mode varies across Node versions per user guidance).
3. Idempotency via `fs.stat` on the target + hash-in-filename: same content → same filename → skip copy.
4. Concurrent-deploy safety: different content produces a different filename, so a live launchd-held executable is never overwritten.
5. No garbage-collection of old versioned copies in this plan — deferred to a future phase (low priority, disk cost is tiny, zero correctness risk).

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| repo → staging dir | Copying `bin/sleepwalker-run-cli` from repo root to `~/.sleepwalker/bin/`. Attack surface: race on file-replace, TOCTOU between stat and copy |
| launchd → staged supervisor | launchd executes `~/.sleepwalker/bin/sleepwalker-run-cli-<hash>`. macOS code-signing + Gatekeeper apply normally; nothing privileged |
| adapter → bundle.bundlePath | bundle.bundlePath flows into TCC warning check + plist WorkingDirectory. Attacker could craft paths containing "/Desktop/" substring to trigger false-positive warnings (non-security) |
| adapter → DeployResult.skillMdContent | Bytes of SKILL.md returned from Desktop deploy flow through to Phase 3 UI. Content is user's own prompt — no cross-boundary trust issue |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation |
|-----------|----------|-----------|-------------|-----------|
| T-02-11-01 | Tampering | supervisor-staging content-hash collision | accept | 8 hex chars = 2^32 address space; collision probability negligible for expected ~dozen supervisor revisions over project lifetime. If production scale demands, bump to 16 chars |
| T-02-11-02 | TOCTOU | supervisor staging race between stat and copyFile | accept | fs.copyFile is atomic (POSIX rename semantics). Stat check is advisory for idempotency, not security-critical. Worst case: two concurrent stages race to copy the same bytes — last write wins, harmless |
| T-02-11-03 | Elevation | staged supervisor in ~/.sleepwalker/ | mitigate | File is owned by deploying user, mode 0o755 (user rwx, group+others rx). No setuid; launchd executes as user. Same trust model as v0.1 bin/sleepwalker-run-cli |
| T-02-11-04 | Info disclosure | skillMdContent in DeployResult | accept | Content is user's own prompt string. Phase 3 UI displays it. No cross-tenant exposure; single-user local dashboard |
| T-02-11-05 | Tampering | TCC path pattern regex bypass | accept | Attacker constructing bundle.bundlePath containing "/Desktop/" substring (e.g. "/usr/Desktop/.../") triggers warning but doesn't affect deploy semantics. Worst case: false-positive warning surfaces |
