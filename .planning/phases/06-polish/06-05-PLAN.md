#!/bin/bash
# tests/compat/frozen-surface.sh
#
# COMP-02: permanent v0.1 frozen-surface regression gate.
# Compares each enumerated path against v0.1 seal commit 998455b.
# Exit 0 if byte-identical OR matches a documented Phase 2-5 additive exception.
# Exit 1 + diff output on any undocumented change.
#
# DO NOT rewrite git history before commit 998455b. Tag v0.1.0 on 998455b if
# repo will undergo significant rebases (Pitfall 2 from Phase 6 RESEARCH §9).

set -euo pipefail

REPO_ROOT="$(cd "$(dirname "$0")/../.." && pwd)"
cd "$REPO_ROOT"

BASELINE="998455b"

# Preflight: baseline commit must exist
if ! git rev-parse --verify "$BASELINE" >/dev/null 2>&1; then
  echo "ERROR: v0.1 baseline commit $BASELINE not in git history."
  echo "Do not rewrite history before this commit."
  echo "If the baseline ref is missing after a legitimate history operation,"
  echo "re-tag it: git tag -a v0.1.0 <new-sha> -m 'v0.1 seal'."
  exit 2
fi

FAIL=0
FAILURES=()

# --- helper: byte-identical check ---
assert_identical() {
  local path="$1"
  if ! git show "$BASELINE:$path" > /tmp/fsurf_base 2>/dev/null; then
    # Path did not exist at baseline — not a Group A path; report misconfiguration
    echo "SKIP: $path not in baseline $BASELINE (report as Group C if intentional)"
    return 0
  fi
  if ! diff -q /tmp/fsurf_base "$path" >/dev/null 2>&1; then
    echo "FAIL (byte-diff): $path changed vs $BASELINE"
    diff /tmp/fsurf_base "$path" | head -20
    FAIL=$((FAIL+1))
    FAILURES+=("$path")
  fi
}

# --- helper: exception check with predicate ---
assert_exception_install_sh() {
  local p="install.sh"
  # v0.1 signature invariants:
  [[ $(grep -c '^set -euo pipefail$' "$p") -eq 1 ]] || { FAILURES+=("$p: set -euo pipefail missing"); FAIL=$((FAIL+1)); return; }
  head -1 "$p" | grep -qE '^#!/bin/(ba)?sh$' || { FAILURES+=("$p: shebang mutated"); FAIL=$((FAIL+1)); return; }
  [[ $(grep -cE "Copying hooks to|Wiring hooks into|Initialize state directory" "$p") -eq 3 ]] || { FAILURES+=("$p: v0.1 install steps count != 3"); FAIL=$((FAIL+1)); return; }
  # Documented Phase 5 amendment present:
  grep -qF 'ERROR: flock is required but not installed' "$p" || { FAILURES+=("$p: flock preflight amendment missing"); FAIL=$((FAIL+1)); return; }
}

assert_exception_audit_log_sh() {
  local p="hooks/sleepwalker-audit-log.sh"
  [[ $(grep -c 'jq -nc' "$p") -eq 1 ]] || { FAILURES+=("$p: jq -nc shape mutated"); FAIL=$((FAIL+1)); return; }
  [[ $(grep -c 'LOCK_FILE=' "$p") -ge 1 ]] || { FAILURES+=("$p: flock amendment missing"); FAIL=$((FAIL+1)); return; }
  # Cross-file invariant: supervisor and hook share the same LOCK_FILE path
  if [[ -f bin/sleepwalker-run-cli ]]; then
    if ! diff <(grep 'LOCK_FILE=' bin/sleepwalker-run-cli) <(grep 'LOCK_FILE=' "$p") >/dev/null 2>&1; then
      FAILURES+=("$p: LOCK_FILE path diverges from bin/sleepwalker-run-cli")
      FAIL=$((FAIL+1))
    fi
  fi
}

assert_exception_queue_ts() {
  local p="dashboard/lib/queue.ts"
  grep -qE "(codex.*gemini|gemini.*codex)" "$p" || { FAILURES+=("$p: QUEU-01 union widen missing"); FAIL=$((FAIL+1)); return; }
}

assert_exception_queue_aggregator_ts() {
  local p="dashboard/lib/queue-aggregator.ts"
  grep -q 'readSupervisorRuns' "$p" || { FAILURES+=("$p: QUEU-03 readSupervisorRuns missing"); FAIL=$((FAIL+1)); return; }
}

assert_exception_package_json() {

}
